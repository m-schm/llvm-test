add(x: int, y: int): int {
	return x + y;
}

factorial(x: int): int {
	y: int = 1;
	while x > 0 {
		y = y * x;
	}
	return y;
}

factorialRecursive(x: int): int {
	if x <= 1 { return 1; }
	return x * factorialRecursive(x - 1);
}

getAge(x: {name: void, age: int}): int {
	return x.age;
}

incrementAge(x: *{name: void, age: int}): int {
	x.age = x.age + 1;
}
